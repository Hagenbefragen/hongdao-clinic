```python
def gregorian_to_13moon(year, month, day):
    """Convert Gregorian date to 13-Moon coordinates."""
    from datetime import date
    
    target = date(year, month, day)
    
    # Determine the 13-Moon year start (Jul 26)
    year_start = date(year, 7, 26)
    if target < year_start:
        year_start = date(year - 1, 7, 26)
    
    day_of_year = (target - year_start).days  # 0-indexed
    
    # Day Out of Time check (day 364 = July 25)
    if day_of_year == 364:
        return {"moon": 0, "day": 0, "name": "Day Out of Time (Green Day)"}
    
    # Handle leap day (Feb 29 in Gregorian is "0.0 Hunab Ku" — outside the count)
    # In leap years, Feb 29 falls within Moon 8 (Galactic Moon)

        try:
            leap_date = date(y, 2, 29)
            if year_start <= leap_date <= target:
                leap_count += 1
        except ValueError:
            pass
    
    adjusted_days = day_of_year - leap_count
```

### 7.2 13-Moon to Gregorian

```python
def moon13_to_gregorian(moon_index, day_index, greg_year):
    """Convert 13-Moon coordinates to Gregorian date."""
    from datetime import date, timedelta
    
    # Determine year start
    year_start = date(greg_year, 7, 26)
    
    if moon_index == 0 and day_index == 0:
        # Day Out of Time
        return date(greg_year + 1, 7, 25)
    
    days_to_add = (moon_index - 1) * 28 + (day_index - 1)
    
    target = year_start + timedelta(days=days_to_add)
    
    # Account for leap days (if Feb 29 exists in range, shift forward)
    for y in range(year_start.year, target.year + 1):
        try:
            leap_date = date(y, 2, 29)
            if year_start <= leap_date <= target:
                target += timedelta(days=1)
        except ValueError:
            pass
    
    return target
```

| **Feb 29 (Leap Day)** | Treated as "0.0 Hunab Ku" — day exists in Gregorian but does NOT advance the 13-Moon count or Kin number |
| **Jul 25 (Day Out of Time)** | Moon = 0, Day = 0. Not part of any Moon or Week. Celebrate as Green Day |

| **Pre-1987 dates** | Kin calculation works backward from the base date (negative elapsed days) |
| **Unicode/Emoji on Windows** | NEVER use emojis in ICS output or terminal print statements — use ASCII-safe descriptions |
| **Google Calendar TRANSPARENT** | ALWAYS set TRANSP:TRANSPARENT to prevent false "busy" blocking |
